실전 예시: AI 공정 어시스턴트

완성된 프로젝트 구조

# app.py (핵심 코드만 발췌)
import streamlit as st
import os
from dotenv import load_dotenv
from utils.ai_helper import GeminiAssistant
from utils.database import ConversationDB
from utils.telegram_bot import send_alert

load_dotenv()

st.set_page_config(page_title="AI 공정 어시스턴트", page_icon="🤖")

# AI 초기화
assistant = GeminiAssistant(api_key=os.getenv("GEMINI_API_KEY"))
db = ConversationDB("data/app.db")

# UI
st.title("🤖 AI 디스플레이 공정 어시스턴트")

# 사이드바
with st.sidebar:
    st.header("설정")
    enable_telegram = st.checkbox("텔레그램 알림 활성화", value=True)
    temperature = st.slider("AI 창의성", 0.0, 1.0, 0.7)

# 메인 영역
tab1, tab2, tab3 = st.tabs(["💬 채팅", "📊 분석", "📜 이력"])

with tab1:
    user_input = st.text_input("질문을 입력하세요:")
    if st.button("전송"):
        with st.spinner("생각 중..."):
            response = assistant.chat(user_input, temperature=temperature)
            st.success(response)

            # DB 저장
            db.save_conversation(user_input, response)

            # 텔레그램 알림 (선택)
            if enable_telegram and "긴급" in response:
                send_alert(f"🚨 {response[:100]}...")

with tab2:
    st.header("📊 데이터 분석")
    # 분석 기능 구현

with tab3:
    st.header("📜 대화 이력")
    history = db.get_recent_conversations(10)
    for item in history:
        st.write(f"**Q:** {item['question']}")
        st.write(f"**A:** {item['answer']}")
        st.divider()
