API = Application Programming Interface

AI 회사의 서버 ↔︎ 내 앱을 연결하는 다리.

내 앱
  │  질문(Request) 전송
  ▼
AI 서버 (Google/Anthropic/OpenAI)
  │  답변(Response) 반환
  ▼
내 앱에 결과 표시

모든 AI API가 이 원리로 작동합니다.

Gemini API 기본 사용법

import google.generativeai as genai
from dotenv import load_dotenv
import os

# 환경변수에서 API Key 로드
load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))

# 모델 초기화
model = genai.GenerativeModel("gemini-1.5-flash")

# 단순 질문-응답
response = model.generate_content(
    "OLED 디스플레이에서 번인 현상이 발생하는 원리를 설명해줘"
)

print(response.text)

실습 앱 — 디스플레이 Q&A 챗봇

import streamlit as st
import google.generativeai as genai
from dotenv import load_dotenv
import os

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-1.5-flash")

st.title("🖥️ 디스플레이 기술 Q&A 챗봇")

SYSTEM_PROMPT = """당신은 디스플레이 기술 전문가입니다.
OLED, LCD, QLED, 수율 관리, 공정 기술에 대해 전문적으로 답변하세요.
삼성디스플레이, LG디스플레이 취업 준비생들의 질문에 친절하게 답변하세요."""

if "messages" not in st.session_state:
    st.session_state.messages = []

for msg in st.session_state.messages:
    with st.chat_message(msg["role"]):
        st.write(msg["content"])

if prompt := st.chat_input("디스플레이 기술 질문을 입력하세요..."):
    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.write(prompt)
    full_prompt = f"{SYSTEM_PROMPT}\n\n질문: {prompt}"
    response = model.generate_content(full_prompt)
    with st.chat_message("assistant"):
        st.write(response.text)
    st.session_state.messages.append(
        {"role": "assistant", "content": response.text}
    )

챗봇 고도화 — 대화 이력 관리

단순 Q&A에서 진짜 대화로 발전시키기:

import google.generativeai as genai

model = genai.GenerativeModel("gemini-1.5-flash")

# 대화 세션 시작 (이전 대화를 자동으로 기억)
chat = model.start_chat(history=[])

# 첫 번째 질문
response1 = chat.send_message("OLED와 LCD의 차이를 설명해줘")
print(response1.text)

# 두 번째 질문 — "그것"이 앞 대화를 가리킴
response2 = chat.send_message("그것의 장단점을 표로 정리해줘")
print(response2.text)  # 앞 대화 맥락 유지

# 대화 이력 확인
for turn in chat.history:
    print(f"[{turn.role}]: {turn.parts[0].text[:50]}...")

Note

model.start_chat(history=[]) — history 파라미터에 이전 대화를 넣으면 대화를 이어갈 수 있습니다. Streamlit의 session_state와 조합하면 완전한 챗봇 완성.

스트리밍 응답 — 타이핑 효과 구현

ChatGPT처럼 글자가 하나씩 나타나는 효과:

import streamlit as st
import google.generativeai as genai

model = genai.GenerativeModel("gemini-1.5-flash")

if prompt := st.chat_input("질문을 입력하세요"):
    with st.chat_message("assistant"):
        # 빈 placeholder 생성
        message_placeholder = st.empty()
        full_response = ""

        # 스트리밍으로 응답 받기
        for chunk in model.generate_content(prompt, stream=True):
            if chunk.text:
                full_response += chunk.text
                # 글자가 쌓이는 효과
                message_placeholder.write(full_response + "▌")

        # 커서 제거 후 최종 출력
        message_placeholder.write(full_response)

Tip

stream=True 옵션 하나로 사용자 경험(UX)이 크게 향상됩니다. 전체 응답을 기다리는 것보다 글자가 나타나는 방식이 훨씬 자연스럽습니다.
