핵심 코드 — 수율 CSV 읽고 알람 보내기

import pandas as pd
import requests
from datetime import datetime

TELEGRAM_TOKEN = "YOUR_TOKEN"
CHAT_ID = "YOUR_CHAT_ID"
THRESHOLD = 95.0

def send_telegram(message: str):
    url = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage"
    requests.post(url, json={
        "chat_id": CHAT_ID,
        "text": message,
        "parse_mode": "Markdown"
    })

def check_yield_and_alert():
    # CSV 읽기 (실제로는 서버나 공유 폴더에서)
    df = pd.read_csv("yield_data.csv")
    latest = df.tail(10)  # 최근 10개 데이터

    below_threshold = latest[latest["yield"] < THRESHOLD]

    if len(below_threshold) > 0:
        lines = below_threshold["line"].tolist()
        avg_yield = below_threshold["yield"].mean()
        message = (
            f"*수율 경고 알람*\n\n"
            f"시각: {datetime.now().strftime('%Y-%m-%d %H:%M')}\n"
            f"평균 수율: {avg_yield:.1f}%\n"
            f"해당 라인: {', '.join(lines)}\n\n"
            f"즉시 확인이 필요합니다."
        )
        send_telegram(message)
    else:
        print(f"[{datetime.now().strftime('%H:%M')}] 수율 정상: "
              f"평균 {latest['yield'].mean():.1f}%")

library(ggplot2)

  Time  = seq(as.POSIXct("2024-01-01 00:00"),
              by = "30 min", length.out = n),
  Yield = 96 + cumsum(rnorm(n, 0, 0.15)),
  Line  = rep(c("Line A", "Line B"), each = 100)
)
yield_df$Yield <- pmin(pmax(yield_df$Yield, 88), 100)
alarm_points <- yield_df[yield_df$Yield < 95, ]

ggplot(yield_df, aes(x = Time, y = Yield, color = Line)) +
  annotate("rect",
           xmin = min(yield_df$Time), xmax = max(yield_df$Time),
           ymin = 88, ymax = 95, alpha = 0.07, fill = "#E74C3C") +
  geom_line(linewidth = 0.8, alpha = 0.9) +
  geom_point(data = alarm_points, aes(x = Time, y = Yield),
             color = "#E74C3C", size = 3, shape = 8) +
  geom_hline(yintercept = 95, linetype = "dashed",
             color = "#E74C3C", linewidth = 1) +
  annotate("text", x = max(yield_df$Time), y = 94.3,
           label = "임계값 95%", color = "#E74C3C",

Gemini API 통합 — 알람에 AI 분석 추가

import google.generativeai as genai
import os

genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-1.5-flash")

def generate_analysis(yield_data: dict) -> str:
    """수율 저하 원인 분석 및 조치 제안"""
    prompt = f"""
    다음 수율 데이터를 분석하고 원인과 조치사항을 제안해주세요:
    - 현재 수율: {yield_data['current_yield']:.1f}%
    - 1시간 전 수율: {yield_data['prev_yield']:.1f}%
    - 저하율: {yield_data['drop_rate']:.1f}%p
    - 해당 라인: {yield_data['line']}
    - 공정 단계: {yield_data.get('process', '미상')}

    간결하게 (3줄 이내) 답변하세요. 불필요한 인사말은 생략하세요.
    """
    response = model.generate_content(prompt)
    return response.text

def check_and_analyze():
    df = pd.read_csv("yield_data.csv")
    latest = df.tail(10)
    below = latest[latest["yield"] < 95.0]

    if len(below) > 0:
        for _, row in below.iterrows():
            analysis = generate_analysis({
                "current_yield": row["yield"],
                "prev_yield": row.get("prev_yield", row["yield"] + 1),
                "drop_rate": row.get("prev_yield", row["yield"] + 1) - row["yield"],
                "line": row["line"]
            })
            msg = (
                f"*AI 수율 분석 알람*\n\n"
                f"라인: {row['line']} | 수율: {row['yield']:.1f}%\n\n"
                f"*Gemini 분석:*\n{analysis}"
            )
            send_telegram(msg)

알람 메시지 레벨 설계

3단계 알람 체계

수율 알람 레벨 행동
97% 이상 정상 없음
95~97% 주의 로그만 기록
92~95% 경고 텔레그램 알람
92% 미만 위험 즉시 전화 + 텔레그램 
def get_alarm_level(yield_pct: float):
    if yield_pct >= 97:
        return "normal", None
    elif yield_pct >= 95:
        return "watch", "INFO"
    elif yield_pct >= 92:
        return "warning", "WARNING"
    else:
        return "critical", "CRITICAL"

from datetime import datetime

def smart_check():
    """교대 근무 패턴에 맞는 스마트 체크"""
    hour = datetime.now().hour

    # 주간 조 (06~18시): 15분마다
    if 6 <= hour < 18:
        interval = 15
    # 야간 조 (18~06시): 5분마다 (더 주의)
    else:
        interval = 5

    check_yield_and_alert()
    # 다음 체크 간격 동적 조정
    schedule.clear("yield_check")
    schedule.every(interval).minutes.do(
        smart_check
    ).tag("yield_check")
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def safe_check():
    """에러 발생해도 스케줄 중단 없이 계속"""
    try:
        check_yield_and_alert()
        logger.info("수율 체크 완료")
    except FileNotFoundError:
        logger.error("CSV 파일을 찾을 수 없음")
        send_telegram("CSV 파일 없음 — 데이터 연결 확인 요망")
    except Exception as e:
        logger.error(f"예상치 못한 오류: {e}")
        send_telegram(f"봇 오류 발생: {str(e)[:100]}")
