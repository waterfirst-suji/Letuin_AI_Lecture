단계 2 (10~35분) — data_simulator.py 생성

Claude Code 프롬프트: “data_simulator.py를 작성해줘.”

# data_simulator.py
import numpy as np
import pandas as pd
from datetime import datetime, timedelta

def simulate_yield(n_points: int = 100) -> pd.DataFrame:
    """수율 데이터 시뮬레이션
    매번 호출할 때마다 약간 다른 데이터 생성
    """
    np.random.seed(int(datetime.now().timestamp()) % 1000)

    times = [
        datetime.now() - timedelta(minutes=5 * i)
        for i in range(n_points, 0, -1)
    ]
    yield_a = np.clip(
        96 + np.cumsum(np.random.normal(0, 0.2, n_points)), 88, 100
    )
    yield_b = np.clip(
        95 + np.cumsum(np.random.normal(0, 0.3, n_points)), 88, 100
    )

    return pd.DataFrame({
        "timestamp": times,
        "Line_A":    yield_a,
        "Line_B":    yield_b
    })

단계 2 계속 — app.py Streamlit 대시보드

Claude Code 프롬프트: “app.py를 작성해줘. KPI 카드와 차트를 포함해줘.”

# app.py
import streamlit as st
import pandas as pd
import plotly.express as px
from data_simulator import simulate_yield
import google.generativeai as genai
import os

genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-1.5-flash")

st.set_page_config(
    page_title="수율 모니터링", page_icon="🏭", layout="wide"
)
st.title("AI 수율 모니터링 대시보드")

# 데이터 로드
if st.button("데이터 새로고침") or "data" not in st.session_state:
    st.session_state.data = simulate_yield(100)

df = st.session_state.data

# KPI 카드 3개
col1, col2, col3 = st.columns(3)
with col1:
    delta_a = df["Line_A"].iloc[-1] - df["Line_A"].iloc[-2]
    st.metric("Line A 현재 수율",
              f"{df['Line_A'].iloc[-1]:.1f}%",
              delta=f"{delta_a:+.1f}%p")
with col2:
    delta_b = df["Line_B"].iloc[-1] - df["Line_B"].iloc[-2]
    st.metric("Line B 현재 수율",
              f"{df['Line_B'].iloc[-1]:.1f}%",
              delta=f"{delta_b:+.1f}%p")
with col3:
    alarm_count = int(
        ((df["Line_A"] < 95) | (df["Line_B"] < 95)).sum()
    )
    st.metric("알람 발생 횟수", alarm_count)

# 차트
fig = px.line(df, x="timestamp", y=["Line_A", "Line_B"],
              title="실시간 수율 추이")
st.plotly_chart(fig, use_container_width=True)

단계 2 완료 — alert_bot.py 알람 봇

# alert_bot.py
import schedule, time, requests, threading
import pandas as pd
from datetime import datetime
from data_simulator import simulate_yield
import os

TELEGRAM_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")
CHAT_ID        = os.getenv("TELEGRAM_CHAT_ID")
THRESHOLD      = 95.0

def send_alert(message: str):
    url = (
        f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage"
    )
    try:
        requests.post(url, json={
            "chat_id":    CHAT_ID,
            "text":       message,
            "parse_mode": "Markdown"
        }, timeout=5)
    except Exception as e:
        print(f"텔레그램 전송 실패: {e}")

def check_and_alert():
    df = simulate_yield(20)  # 최근 20개 포인트
    for line in ["Line_A", "Line_B"]:
        current = df[line].iloc[-1]
        if current < THRESHOLD:
            send_alert(
                f"*수율 경고*\n"
                f"라인: {line}\n"
                f"수율: {current:.1f}%\n"
                f"시각: {datetime.now().strftime('%H:%M')}"
            )

def run_scheduler():
    """별도 스레드에서 스케줄러 실행"""
    schedule.every(5).minutes.do(check_and_alert)
    while True:
        schedule.run_pending()
        time.sleep(10)

# Streamlit 앱 시작 시 백그라운드 스레드로 실행
bot_thread = threading.Thread(target=run_scheduler, daemon=True)
bot_thread.start()

완성된 시스템 — 실시간 수율 모니터링

library(ggplot2)

set.seed(42)
n <- 60
timeline <- data.frame(
  Minute  = 1:n,
  Yield_A = pmin(pmax(96 + cumsum(rnorm(n,  0.0, 0.20)), 88), 100),
  Yield_B = pmin(pmax(95 + cumsum(rnorm(n, -0.05, 0.25)), 88), 100)
)
alarm_pts_a <- timeline[timeline$Yield_A < 95, ]
alarm_pts_b <- timeline[timeline$Yield_B < 95, ]
all_alarm <- unique(c(alarm_pts_a$Minute, alarm_pts_b$Minute))

ggplot(timeline, aes(x = Minute)) +
  annotate("rect", xmin = 0, xmax = 60, ymin = 88, ymax = 95,
           alpha = 0.07, fill = "#E74C3C") +
  geom_line(aes(y = Yield_A, color = "Line A"), linewidth = 1) +
  geom_line(aes(y = Yield_B, color = "Line B"), linewidth = 1) +
  geom_hline(yintercept = 95, linetype = "dashed",
             color = "#E74C3C", linewidth = 1) +
  {if (length(all_alarm) > 0)
    geom_vline(xintercept = all_alarm,
               color = "#E74C3C", alpha = 0.4, linewidth = 0.8)} +
  scale_color_manual(values = c("Line A" = "#3498DB", "Line B" = "#27AE60")) +
  scale_y_continuous(labels = function(x) paste0(x, "%")) +
  theme_minimal(base_size = 13) +
  labs(
    title = "통합 시스템: 실시간 수율 모니터링 (빨간 수직선 = 알람 발생)",
    x = "경과 시간 (분)", y = "수율 (%)", color = "라인"
  )
