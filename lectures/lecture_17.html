# telegram_test.py
import requests

BOT_TOKEN = "YOUR_BOT_TOKEN"
CHAT_ID = "YOUR_CHAT_ID"

def send_message(text):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    data = {
        "chat_id": CHAT_ID,
        "text": text
    }
    response = requests.post(url, data=data)
    return response.json()

# 테스트
result = send_message("🤖 안녕하세요! 디스플레이 알림 봇입니다.")
print(result)

실전 예제: 수율 알림 봇

# yield_alert_bot.py
import requests
import pandas as pd
from datetime import datetime

BOT_TOKEN = "YOUR_BOT_TOKEN"
CHAT_ID = "YOUR_CHAT_ID"

def send_alert(message):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    data = {"chat_id": CHAT_ID, "text": message, "parse_mode": "Markdown"}
    requests.post(url, data=data)

def check_yield():
    # 가상의 수율 데이터
    yield_rate = 92.3  # 실제로는 DB나 API에서 가져옴
    threshold = 95.0

    if yield_rate < threshold:
        alert_message = f"""
⚠️ **수율 이상 감지**

📊 현재 수율: {yield_rate}%
🎯 목표 수율: {threshold}%
📉 편차: {threshold - yield_rate:.1f}%

🕐 시각: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}
"""
        send_alert(alert_message)
        print("✅ 알림 전송 완료")
    else:
        print("✅ 수율 정상")

if __name__ == "__main__":
    check_yield()

고급 기능 1: 명령어 처리

사용자가 봇에게 명령을 내리면 자동 응답하는 기능

# interactive_bot.py
from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes

BOT_TOKEN = "YOUR_BOT_TOKEN"

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "안녕하세요! 사용 가능한 명령어:\n"
        "/status - 현재 공정 상태\n"
        "/yield - 오늘 수율\n"
        "/help - 도움말"
    )

async def status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("📊 공정 상태: 정상 가동 중")

async def yield_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("📈 오늘 수율: 96.2%")

# 봇 실행
app = Application.builder().token(BOT_TOKEN).build()
app.add_handler(CommandHandler("start", start))
app.add_handler(CommandHandler("status", status))
app.add_handler(CommandHandler("yield", yield_cmd))
app.run_polling()

고급 기능 2: 이미지 전송

# send_chart_bot.py
import requests
import matplotlib.pyplot as plt
import io

BOT_TOKEN = "YOUR_BOT_TOKEN"
CHAT_ID = "YOUR_CHAT_ID"

def create_yield_chart():
    # 수율 데이터 시각화
    days = ['월', '화', '수', '목', '금']
    yields = [95.2, 96.1, 94.8, 97.3, 96.5]

    plt.figure(figsize=(8, 5))
    plt.plot(days, yields, marker='o', linewidth=2)
    plt.title('주간 수율 트렌드', fontsize=14)
    plt.ylabel('수율 (%)')
    plt.grid(True, alpha=0.3)

    # 이미지를 바이트로 변환
    buf = io.BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight')
    buf.seek(0)
    return buf

def send_photo(caption):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendPhoto"
    files = {'photo': create_yield_chart()}
    data = {'chat_id': CHAT_ID, 'caption': caption}
    requests.post(url, files=files, data=data)

send_photo("📊 이번 주 수율 리포트")

실전 배포: AWS Lambda + Cron

매일 오전 9시에 자동으로 수율 리포트 전송

# lambda_function.py (AWS Lambda)
import json
import requests
import os

def lambda_handler(event, context):
    BOT_TOKEN = os.environ['TELEGRAM_BOT_TOKEN']
    CHAT_ID = os.environ['TELEGRAM_CHAT_ID']

    # 일일 리포트 생성
    message = f"""
📊 **일일 생산 리포트**

날짜: 2026-04-11
수율: 96.2%
생산량: 10,250 units
불량률: 3.8%

✅ 정상 가동 중
"""

    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    data = {"chat_id": CHAT_ID, "text": message, "parse_mode": "Markdown"}
    response = requests.post(url, data=data)

    return {
        'statusCode': 200,
        'body': json.dumps('Message sent!')
    }
