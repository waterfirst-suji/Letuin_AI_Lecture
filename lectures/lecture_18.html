STEP 1: 가상 데이터 생성

# generate_sensor_data.py
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

def generate_fake_data(hours=24):
    """가상의 센서 데이터 생성"""
    timestamps = [datetime.now() - timedelta(hours=i) for i in range(hours)]

    data = pd.DataFrame({
        'timestamp': timestamps,
        'temperature': np.random.normal(250, 5, hours),  # 평균 250°C
        'pressure': np.random.normal(1.2, 0.1, hours),   # 평균 1.2 bar
        'yield_rate': np.random.normal(96, 2, hours)     # 평균 96%
    })

    # 일부러 이상 데이터 주입 (10% 확률)
    anomaly_idx = np.random.choice(hours, size=int(hours*0.1), replace=False)
    data.loc[anomaly_idx, 'yield_rate'] = np.random.uniform(88, 93, len(anomaly_idx))

    return data.sort_values('timestamp')

# 데이터 생성 및 저장
df = generate_fake_data()
df.to_csv('sensor_data.csv', index=False)
print("✅ 가상 센서 데이터 생성 완료")

실습: 이상 감지 로직 (10분)

# anomaly_detector.py
import pandas as pd
from datetime import datetime

class AnomalyDetector:
    def __init__(self, thresholds):
        self.thresholds = thresholds
        self.alert_count = {}  # 연속 알림 방지

    def check_anomaly(self, row):
        """이상 감지"""
        anomalies = []

        # 수율 체크
        if row['yield_rate'] < self.thresholds['yield_min']:
            anomalies.append(f"수율 저하: {row['yield_rate']:.1f}% (기준: {self.thresholds['yield_min']}%)")

        # 온도 체크
        if row['temperature'] > self.thresholds['temp_max']:
            anomalies.append(f"온도 초과: {row['temperature']:.1f}°C (기준: {self.thresholds['temp_max']}°C)")

        # 압력 체크
        if row['pressure'] > self.thresholds['pressure_max']:
            anomalies.append(f"압력 초과: {row['pressure']:.2f} bar (기준: {self.thresholds['pressure_max']} bar)")

        return anomalies

    def should_alert(self, sensor_id, anomalies):
        """연속 알림 방지 (5분 간격)"""
        if not anomalies:
            return False

        now = datetime.now()
        last_alert = self.alert_count.get(sensor_id)

        if last_alert is None or (now - last_alert).seconds > 300:
            self.alert_count[sensor_id] = now
            return True

        return False

실습: 텔레그램 알림 통합 (20분)

# monitoring_system.py
import pandas as pd
import requests
import time
from datetime import datetime
from anomaly_detector import AnomalyDetector
import os
from dotenv import load_dotenv

load_dotenv()

BOT_TOKEN = os.getenv("TELEGRAM_BOT_TOKEN")
CHAT_ID = os.getenv("TELEGRAM_CHAT_ID")

def send_alert(anomalies, row):
    """텔레그램 알림 전송"""
    message = f"""
🚨 **공정 이상 감지**

⏰ 시각: {row['timestamp']}

⚠️ **이상 항목:**
{''.join([f'• {a}' for a in anomalies])}

📊 **현재 상태:**
• 온도: {row['temperature']:.1f}°C
• 압력: {row['pressure']:.2f} bar
• 수율: {row['yield_rate']:.1f}%

👉 즉시 확인 필요
"""

    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
    data = {"chat_id": CHAT_ID, "text": message, "parse_mode": "Markdown"}
    response = requests.post(url, data=data)
    return response.status_code == 200

def monitor_loop():
    """실시간 모니터링 루프"""
    detector = AnomalyDetector({
        'yield_min': 95.0,
        'temp_max': 260.0,
        'pressure_max': 1.4
    })

    print("🔍 모니터링 시작...")

    while True:
        # CSV에서 최신 데이터 읽기
        df = pd.read_csv('sensor_data.csv')
        latest = df.iloc[-1]

        # 이상 감지
        anomalies = detector.check_anomaly(latest)

        # 알림 필요 여부 확인
        if detector.should_alert('main_sensor', anomalies):
            if send_alert(anomalies, latest):
                print(f"✅ 알림 전송: {len(anomalies)}개 이상 감지")
            else:
                print("❌ 알림 전송 실패")
        else:
            print(f"✅ {latest['timestamp']} - 정상")

        time.sleep(5)  # 5초마다 체크

if __name__ == "__main__":
    monitor_loop()

실습: Streamlit 대시보드 통합 (30분)

# dashboard.py
import streamlit as st
import pandas as pd
import plotly.express as px
from datetime import datetime, timedelta

st.set_page_config(page_title="공정 감시 시스템", page_icon="🏭", layout="wide")

# 제목
st.title("🏭 디스플레이 공정 감시 시스템")
st.markdown("---")

# 데이터 로드
@st.cache_data(ttl=5)  # 5초마다 자동 갱신
def load_data():
    return pd.read_csv('sensor_data.csv', parse_dates=['timestamp'])

df = load_data()

# 최신 상태 표시
col1, col2, col3 = st.columns(3)
latest = df.iloc[-1]

with col1:
    st.metric("🌡️ 온도", f"{latest['temperature']:.1f}°C",
              delta=f"{latest['temperature'] - df.iloc[-2]['temperature']:.1f}°C")

with col2:
    st.metric("💨 압력", f"{latest['pressure']:.2f} bar",
              delta=f"{latest['pressure'] - df.iloc[-2]['pressure']:.2f} bar")

with col3:
    yield_delta = latest['yield_rate'] - df.iloc[-2]['yield_rate']
    st.metric("📊 수율", f"{latest['yield_rate']:.1f}%",
              delta=f"{yield_delta:.1f}%",
              delta_color="normal" if yield_delta >= 0 else "inverse")

# 시계열 그래프
st.markdown("### 📈 실시간 트렌드 (최근 24시간)")

fig = px.line(df, x='timestamp', y='yield_rate',
              title='수율 트렌드',
              labels={'timestamp': '시각', 'yield_rate': '수율 (%)'})
fig.add_hline(y=95, line_dash="dash", line_color="red",
              annotation_text="임계값 (95%)")
st.plotly_chart(fig, use_container_width=True)

# 이상 감지 로그
st.markdown("### 🚨 최근 알림 이력")
anomalies = df[df['yield_rate'] < 95].tail(10)
if len(anomalies) > 0:
    st.dataframe(anomalies[['timestamp', 'temperature', 'pressure', 'yield_rate']],
                 use_container_width=True)
else:
    st.success("✅ 최근 이상 없음")

# 자동 갱신
st.button("🔄 새로고침", on_click=st.cache_data.clear)

고급: 예측 알림 시스템

Prophet 라이브러리로 수율을 예측하고, 하락이 예상되면 사전 알림

# predictive_alert.py
from prophet import Prophet
import pandas as pd

def forecast_yield(df):
    """수율 예측"""
    # Prophet 형식으로 변환
    prophet_df = df[['timestamp', 'yield_rate']].rename(
        columns={'timestamp': 'ds', 'yield_rate': 'y'}
    )

    # 모델 학습
    model = Prophet(interval_width=0.95)
    model.fit(prophet_df)

    # 향후 6시간 예측
    future = model.make_future_dataframe(periods=6, freq='H')
    forecast = model.predict(future)

    # 예측값이 임계값 이하인지 확인
    next_6h = forecast.tail(6)
    critical = next_6h[next_6h['yhat'] < 95.0]

    if len(critical) > 0:
        alert_time = critical.iloc[0]['ds']
        predicted_yield = critical.iloc[0]['yhat']

        message = f"""
⚠️ **수율 하락 예측**

🔮 예상 시각: {alert_time}
📉 예측 수율: {predicted_yield:.1f}%
🎯 임계값: 95.0%

💡 사전 조치 권장
"""
        send_alert_message(message)
