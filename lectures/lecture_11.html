  x     = c(5,   2,   8,   1,   3,    6,     9),
  y     = c(4,   3,   3,   2,   2,    2,     2),
  label = c("새 앱\n만들기",

            "Heroku\n/Render",

            "Vercel\n/Netlify"),

)

edges_dt <- data.frame(
  x    = c(5,5,2,2,8,8),
  xend = c(2,8,1,3,6,9),
  y    = c(4,4,3,3,3,3),
  yend = c(3,3,2,2,2,2),
  label= c("YES","NO","YES","NO","YES","NO")
)

ggplot() +
  geom_segment(data=edges_dt,
               aes(x=x, xend=xend, y=y, yend=yend),
               color="gray60", linewidth=1,
               arrow=arrow(length=unit(0.15,"inches"), type="closed")) +
  geom_label(data=edges_dt,
             aes(x=(x+xend)/2, y=(y+yend)/2, label=label),
             size=2.8, fill="gray20", color="white") +
  geom_point(data=nodes_dt,
             aes(x=x, y=y, color=type), size=14) +

            aes(x=x, y=y, label=label),

Netlify — JAMstack의 왕좌

Netlify란?

JAMstack (JavaScript + API + Markup) 특화 플랫폼

강점: - 드래그 앤 드롭으로 HTML 폴더 배포 - 자동 HTTPS - 폼 처리 (백엔드 없이 문의 폼 구현) - A/B 테스트 기능 - 분기별 프리뷰 URL

# 방법 1: CLI 배포
npm install -g netlify-cli
netlify deploy --dir=./dist --prod

# 방법 2: 드래그 앤 드롭
# app.netlify.com에 폴더 드래그

Netlify 무료 플랜

항목 한도
빌드 시간 300분/월
대역폭 100GB/월
사이트 수 무제한
팀원 수 1명

디스플레이 업 활용

제품 소개 사이트, 스펙 비교 페이지, 이력서/포트폴리오에 최적. GitHub + Netlify = 자동 배포 파이프라인.

Hugging Face Spaces — AI 모델 배포의 성지

Replit — 브라우저에서 바로 코딩하고 배포

설치 없이 바로 시작

Replit의 장점: - 브라우저만 있으면 개발 가능 - 50+ 언어 지원 (Python, JS, R, …) - 배포 버튼 한 번으로 공개 - 실시간 협업 (구글독스처럼) - AI 코드 어시스턴트 내장

플랫폼 속도 비교 — 실제 로딩 타임

library(ggplot2)

load_times <- data.frame(
  Platform = factor(
    c("GitHub Pages","Vercel","Netlify","Streamlit","Render","Hugging Face"),
    levels=c("GitHub Pages","Vercel","Netlify","Streamlit","Render","Hugging Face")
  ),
  Cold_Start = c(0.3, 0.5, 0.4, 8.0, 12.0, 15.0),
  Warm_Start = c(0.3, 0.5, 0.4,  2.0,  1.5,  2.5),
  Type       = c("정적","정적","정적","동적","동적","동적")
)

load_long <- reshape(load_times,
  varying=c("Cold_Start","Warm_Start"),
  v.names="Seconds",
  timevar="StartType",
  times=c("콜드 스타트 (첫 방문)","웜 스타트 (재방문)"),
  direction="long"
)

ggplot(load_long, aes(x=Platform, y=Seconds, fill=StartType)) +
  geom_bar(stat="identity", position="dodge", width=0.65) +
  scale_fill_manual(values=c("콜드 스타트 (첫 방문)"="#E74C3C",
                              "웜 스타트 (재방문)"="#27AE60")) +
  scale_y_continuous(labels=function(x) paste0(x,"초")) +
  theme_minimal(base_size=12) +
  labs(title="플랫폼별 페이지 로딩 시간 비교 (초)",
       subtitle="정적 사이트 0.3~0.5초 vs 동적 앱 콜드스타트 8~15초 차이 주의",
       x="", y="로딩 시간 (초)", fill="")
